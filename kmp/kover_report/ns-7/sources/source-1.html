     suspend fun getPokemonList(
         offset: Int,
         limit: Int
     ): PokeApiPaginatedResponse<PokemonItemResponse> {
         val response = pokeApiClient.client.get("/api/v2/pokemon?limit=${limit}&offset=${offset}")

     }
 
     suspend fun getPokemon(id: String): PokemonResponse {

     }
